## 3. DB 스키마 추가

기존 `ServiceApplication` 모델 확장:

```prisma
model ServiceApplication {
  // ... 기존 필드 ...
  
  // 도입 진행 추적 (신규)
  assignedSalesId  BigInt?   @map("assigned_sales_id")     // 영업 담당자
  assignedPmId     BigInt?   @map("assigned_pm_id")        // 도입 PM
  
  // 단계별 완료 시각 (각 Day별 milestone)
  appliedAt        DateTime  @default(now())  // Day 0
  reviewStartedAt  DateTime?                  // Day 1
  quotedAt         DateTime?                  // Day 2
  meetingAt        DateTime?                  // Day 3-4
  contractedAt     DateTime?                  // Day 5
  instanceReadyAt  DateTime?                  // Day 6-7
  dataReadyAt      DateTime?                  // Day 8-10
  trainedAt        DateTime?                  // Day 11-12
  pilotStartedAt   DateTime?                  // Day 13
  goLiveAt         DateTime?                  // Day 14
  
  // 지연 추적
  expectedGoLive   DateTime?                  // 예상 운영 개시일 (자동 계산)
  delayDays        Int       @default(0)
  delayReason      String?   @db.Text
  
  // 메모
  notes            ApplicationNote[]
  
  // ...
}

model ApplicationNote {
  id             BigInt   @id @default(autoincrement())
  applicationId  BigInt   @map("application_id")
  authorId       BigInt   @map("author_id")
  authorName     String   @db.VarChar(50)
  content        String   @db.Text
  type           NoteType @default(GENERAL)
  createdAt      DateTime @default(now()) @map("created_at")
  
  application    ServiceApplication @relation(fields: [applicationId], references: [id], onDelete: Cascade)
  
  @@index([applicationId, createdAt])
  @@map("application_notes")
}

enum NoteType {
  GENERAL          // 일반 메모
  DELAY_REASON     // 지연 사유
  CUSTOMER_REQUEST // 고객 요청
  ESCALATION       // 에스컬레이션
}
```

## 5. 지연 자동 감지

### 5.1 표준 일정 매핑

```typescript
// lib/onboarding-schedule.ts

const STANDARD_DAYS = {
  applied: 0,
  review_started: 1,
  quoted: 2,
  meeting: 4,
  contracted: 5,
  instance_ready: 7,
  data_ready: 10,
  trained: 12,
  pilot_started: 13,
  go_live: 14,
} as const;

export function calculateExpectedGoLive(appliedAt: Date): Date {
  const d = new Date(appliedAt);
  // 14 영업일 (주말 제외)
  let workDays = 0;
  while (workDays < 14) {
    d.setDate(d.getDate() + 1);
    const day = d.getDay();
    if (day !== 0 && day !== 6) workDays++;
  }
  return d;
}

export function calculateDelayDays(application: ServiceApplication): number {
  const now = new Date();
  if (application.status === 'CONVERTED') return 0;
  
  const expected = application.expectedGoLive ?? calculateExpectedGoLive(application.appliedAt);
  if (now <= expected) return 0;
  
  // 영업일 기준 지연
  let delayDays = 0;
  const d = new Date(expected);
  while (d <= now) {
    const day = d.getDay();
    if (day !== 0 && day !== 6) delayDays++;
    d.setDate(d.getDate() + 1);
  }
  return Math.max(0, delayDays - 1);
}
```

### 5.2 일일 cron — 지연 감지 + 알림

```typescript
// app/api/cron/onboarding-delay-check/route.ts

export async function POST(req: Request) {
  // Bearer 토큰 검증
  if (req.headers.get('authorization') !== `Bearer ${process.env.CRON_SECRET}`) {
    return NextResponse.json({ error: 'unauthorized' }, { status: 401 });
  }
  
  const inProgress = await prisma.serviceApplication.findMany({
    where: {
      status: { in: ['SUBMITTED', 'UNDER_REVIEW', 'QUOTED', 'CONTRACTED', 'ONBOARDING'] },
    },
  });
  
  let delayedCount = 0;
  for (const app of inProgress) {
    const delay = calculateDelayDays(app);
    if (delay > app.delayDays) {
      // 새로 지연 발생 또는 추가 지연
      await prisma.serviceApplication.update({
        where: { id: app.id },
        data: { delayDays: delay },
      });
      
      if (delay >= 2) {
        // Slack 알림 — 2일 이상 지연
        await sendSlackAlert(`⚠️ ${app.companyName} (${app.applicationNo}) ${delay}일 지연 — 단계: ${app.status}`);
        delayedCount++;
      }
    }
  }
  
  return NextResponse.json({ ok: true, checked: inProgress.length, newDelays: delayedCount });
}
```

`vercel.json`:
```json
{ "crons": [{ "path": "/api/cron/onboarding-delay-check", "schedule": "0 23 * * *" }] }
```

매일 23:00 UTC = 08:00 KST 실행 → 출근 직후 운영팀 알림.

```tsx
function KanbanBoard({ cards }: { cards: Card[] }) {
```

## 7. Bottleneck 분석 (선택 — Phase 2)

### 7.1 분석 화면 `/super-admin/onboarding/insights`

```sql
-- 단계별 평균 소요일
SELECT
  AVG(EXTRACT(EPOCH FROM (review_started_at - applied_at)) / 86400) AS avg_review_days,
  AVG(EXTRACT(EPOCH FROM (quoted_at - review_started_at)) / 86400) AS avg_quote_days,
  AVG(EXTRACT(EPOCH FROM (contracted_at - quoted_at)) / 86400) AS avg_contract_days,
  AVG(EXTRACT(EPOCH FROM (go_live_at - contracted_at)) / 86400) AS avg_implementation_days
FROM service_applications
WHERE status = 'CONVERTED'
  AND applied_at >= NOW() - INTERVAL '90 days';
```

→ "검토→견적" 평균 3일이면 표준 1일에서 200% 초과 → 영업팀 인력 보강 의사결정 자료.

### 7.2 영업 담당자별 성과

```sql
SELECT
  u.name AS sales,
  COUNT(*) AS total,
  COUNT(*) FILTER (WHERE sa.status = 'CONVERTED') AS converted,
  AVG(sa.delay_days) AS avg_delay
FROM service_applications sa
JOIN users u ON u.id = sa.assigned_sales_id
GROUP BY u.id, u.name
ORDER BY converted DESC;
```
